## 💻 3️⃣ CMake 구성 요약

```cmake
option(USE_ORT "Enable ONNX Runtime backend" OFF)

if(USE_ORT)
    set(ONNXRUNTIME_ROOT "D:/SDKs/onnxruntime-gpu-1.18.0")
    add_library(onnxruntime SHARED IMPORTED)
    set_target_properties(onnxruntime PROPERTIES
        IMPORTED_IMPLIB "${ONNXRUNTIME_ROOT}/lib/onnxruntime.lib"
        IMPORTED_LOCATION "${ONNXRUNTIME_ROOT}/bin/onnxruntime.dll"
    )
    target_include_directories(onnxruntime INTERFACE "${ONNXRUNTIME_ROOT}/include")
    target_link_libraries(boatcnt PRIVATE onnxruntime)
    target_sources(boatcnt PRIVATE src/detector_onnxruntime.cpp)
else()
    target_sources(boatcnt PRIVATE src/detector_stub.cpp)
endif()
```

📌 기본 모드 → STUB만 빌드 (`USE_ORT=OFF`)  
📌 ONNX Runtime 준비 시 → `USE_ORT=ON` 으로 전환
